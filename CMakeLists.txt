cmake_minimum_required(VERSION 3.10)\n\n# Project Name\nproject(Niell)\n\n# Define the output DLL name\nset(OUTPUT_NAME niell)\n\n# Add the source files\nfile(GLOB SOURCES src/*.cpp)\n\n# Find the required packages\nfind_package(DirectX REQUIRED)\nfind_package(OpenGL REQUIRED)\nfind_package(imgui REQUIRED)\n\n# Create the DLL target\nadd_library(${OUTPUT_NAME} SHARED ${SOURCES})\n\n# Set the output directory\nset_target_properties(${OUTPUT_NAME} PROPERTIES\n    LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/release\n)\n\n# Include directories for DirectX and ImGui\ninclude_directories(${DIRECTX_INCLUDE_DIRS})\ninclude_directories(${IMGUI_INCLUDE_DIRS})\n\n# Link the necessary libraries\ntarget_link_libraries(${OUTPUT_NAME} ${DIRECTX_LIBRARIES} ${OPENGL_LIBRARIES} ${IMGUI_LIBRARIES})\n\n# Set compiler flags for Windows DLL output\nif(MSVC)\n    target_compile_options(${OUTPUT_NAME} PRIVATE /W4 /DWIN32_LEAN_AND_MEAN)\nelse()\n    target_compile_options(${OUTPUT_NAME} PRIVATE -Wall -Wextra)\nendif()\n
